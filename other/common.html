import settings from 'rh-lib/src/settings'

computed: {
	rhMarginBase() {
	  	const rhMarginBase = getComputedStyle(document.documentElement).getPropertyValue('--rh-margin-base')
	  	return parseFloat(rhMarginBase) * settings.baseSize
	}
}
